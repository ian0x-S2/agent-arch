# ARCH POLICY v1.0.0
STACK: frontend | feature-sliced | state=feature-based | style=utility-first | routing=declarative

LAYERS & RESPONSIBILITIES:
  APP: [pages, widgets, features, entities, shared]
    OWNS: providers, routing, global styles, app initialization
    NOT: contain business logic
  PAGES: [widgets, features, entities, shared]
    OWNS: composition of widgets for a route
    NOT: contain business logic
  WIDGETS: [features, entities, shared]
    OWNS: composition of features, reusable page sections
    NOT: contain business logic directly
  FEATURES: [entities, shared]
    OWNS: user interactions with business value (AddToCart, LoginForm)
    NOT: import from other features, know about pages
  ENTITIES: [shared]
    OWNS: business objects and their operations (User, Product, Order)
    NOT: import from features or above, contain UI components ideally
  SHARED: []
    OWNS: reusable infra with no business logic (ui-kit, api client, utils)
    NOT: import from any other layer, contain business logic
  circular=FORBIDDEN | cross_feature=via-public-api-only

ABSTRACTION BOUNDARIES:
  features→entities: entities (inner) ← features (outer) [API_REQ=true]

DOMAIN RULES: Entity location: entities | No framework: true | Anemic allowed: false

EXPECTED STRUCTURE:
  FSD: src/{app, pages, widgets, features, entities, shared}/<slice>/{ui,model,api,lib,config}

FILE CONVENTIONS:
  component=*.component.tsx companions=[test(required)]
  hook=*.hook.ts companions=[test(required)]
  store=*.store.ts companions=[test(required)]
  service=*.service.ts companions=[test(optional)]
  types=*.types.ts companions=[none]
  constants=*.constants.ts companions=[none]
  colocation=strict | test=colocated | public_api=REQUIRED(index.ts)
